42 const uint8_t* b = reinterpret_cast<const uint8_t*>(0X20000000UL);
43 uint8_t* d = reinterpret_cast<uint8_t*>(dst);
44 const uint8_t *s = reinterpret_cast<const uint8_t*>(src);

46 !((d < b && b <= (d + len)) || (s < b && b <= (s + len)))) {
