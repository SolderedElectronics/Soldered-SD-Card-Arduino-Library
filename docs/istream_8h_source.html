367void istream::getNumber(T* value) {
368 uint32_t tmp;
369 if ((T)-1 < 0) {
370 // number is signed, max positive value
371 uint32_t const m = ((uint32_t)-1) >> (33 - sizeof(T) * 8);

374 *value = (T)tmp;
375 }
376 } else {
377 // max unsigned value for T
